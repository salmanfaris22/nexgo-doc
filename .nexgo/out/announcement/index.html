Auth Docs →
Auth Setup 
import (
    "github.com/salmanfaris22/nexgo/pkg/auth"
    "github.com/salmanfaris22/nexgo/pkg/session"
)

// Configure JWT
jwtCfg := auth.JWTConfig{
    Secret:    []byte("my-secret-key"),
    ExpiresIn: 24 * time.Hour,
    Issuer:    "myapp",
}

// Sign a token
token, _ := auth.Sign(jwtCfg, auth.Claims{
    UserID: "usr_123",
    Role:   "admin",
})

// Protect routes
app.Group("/dashboard",
    auth.RequireJWT(jwtCfg),
    auth.RequireRole("admin"),
)

// Sessions
store := session.NewCookieStore("session-secret")
sess, _ := store.Get(r, "sid")
sess.Set("user", "usr_123")
sess.Flash("Welcome back!")
